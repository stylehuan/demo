模拟PlaceHolder

例子1

输入框占位效果

            var text = new $.IC.PlaceHolder('#userName',{
                text: "请输入您的用户名"
            });

例子2

textarea占位效果

可用 不可用 

            var text = new $.IC.PlaceHolder('#textarea',{
                text: "请输入您的用户名"
            });

            $("#enable").on("click", function(){
                textarea.enable();
                return false;
            });
            $("#disable").on("click", function(){
                textarea.disable();
                return false;
            });
